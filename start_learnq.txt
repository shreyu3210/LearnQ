### LearnQ Infrastructure Commands

# 1. Start PostgreSQL Database
# Matches .env: DATABASE_URL="postgresql://postgres:123456@localhost/learnQ"

docker start learnq-db

docker run --name learnq-db \
  -e POSTGRES_USER=postgres \
  -e POSTGRES_PASSWORD=123456 \
  -e POSTGRES_DB=learnQ \
  -v ~/postgres_data:/var/lib/postgresql/data \
  -p 5432:5432 \
  -d postgres:15

# 2. Start Triton Inference Server (24.12)
# Ensure paths match your Linux home directory
docker run --gpus all -it --rm \
  -p 8000:8000 -p 8001:8001 -p 8002:8002 \
  --shm-size=1g \
  -v /home/shreyansh1812/BE/LearnQ/Test_triton/model_repository:/models \
  nvcr.io/nvidia/tritonserver:24.12-py3 \
  tritonserver --model-repository=/models --strict-model-config=false

# 3. Shutdown Commands
# Stop FastAPI: Ctrl+C in its terminal
# Stop Triton: Ctrl+C in its terminal
# Stop Postgres:
docker stop learnq-db
